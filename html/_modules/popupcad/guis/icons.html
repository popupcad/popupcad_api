import os
import glob
import qt
qc = qt.QtCore
qg = qt.QtGui
import popupcad

filenames = glob.glob(os.path.join(popupcad.supportfiledir,'icons','*.png'))


[docs]def build():
    icons = {}
    for filename in filenames:
        key = os.path.split(filename)[1]
        key = os.path.splitext(key)[0]
        icons[key] = qg.QIcon(filename)
    return icons
