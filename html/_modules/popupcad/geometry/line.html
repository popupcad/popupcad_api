[docs]    def constraints_ref(self):
        try:
            return self._constraints_ref
        except AttributeError:
            self._constraints_ref = SymbolicLine(
                self.vertex1.constraints_ref(),
                self.vertex2.constraints_ref())
            return self._constraints_ref
