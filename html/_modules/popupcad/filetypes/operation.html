        if dialog.exec_() == dialog.Accepted:
            self.editdata(*dialog.acceptdata())
            editedsignal.emit(self)

        qc = qt.QtCore
        qg = qt.QtGui
        result, ok = qg.QInputDialog.getText(
            None, 'description', 'label', text=self.description)
        if ok:
            self.description = result
