        self.index -= 1
        if self.index < 0:
            self.index = 0
        self.load_method(self.undoqueue[self.index])



[docs]    def redo(self):
        self.index += 1
        m = len(self.undoqueue)
        if self.index >= m:
            self.index = m - 1
        self.load_method(self.undoqueue[self.index])
