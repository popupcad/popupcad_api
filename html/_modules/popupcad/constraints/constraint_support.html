[docs]    def p(self):
        p_x = Variable(str(self) + '_x')
        p_y = Variable(str(self) + '_y')
        return sympy.Matrix([p_x, p_y, 0])


    def __hash__(self):
        return self.id

    def __eq__(self, other):
        if isinstance(self, type(other)):
            return self.id == other.id
        return False

    def __lt__(self, other):
        return self.id < other.id

    def __str__(self):
        return 'vertex' + str(self.id)
